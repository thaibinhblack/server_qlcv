CREATE OR REPLACE FUNCTION HGI_GET_THONGTIN_DK (
    p_TIEU_CH_SEARCH VARCHAR2,
    P_MA_HO_SO VARCHAR2
)
RETURN VARCHAR2 AS 
    RESULT_FC VARCHAR2(5000);

BEGIN
    SELECT GIA_TRI INTO RESULT_FC  from hs_du_lieu_ho_so where ma_tieu_chi_search  = p_TIEU_CH_SEARCH and ma_ho_so = P_MA_HO_SO;
    RETURN RESULT_FC;
END HGI_GET_THONGTIN_DK;
